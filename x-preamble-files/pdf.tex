\usepackage{ifpdf} % Allows for checking whether file will be output as PDF

\ifpdf
     \usepackage{graphicx}% [pdftex]
     \usepackage[pdftex,colorlinks=true,linkcolor=black,citecolor=black]{hyperref}
\else
     \usepackage[dvips]{graphicx}
     \usepackage[dvips,colorlinks=true,linkcolor=black,citecolor=black]{hyperref}
\fi

\hypersetup{pdftitle    = {\TITLE},
            pdfsubject  = {\SUBJECT},
            pdfkeywords = {\KEYWORDS},
            pdfauthor   = { \textcopyright Layout by Jonas Sigel, used by: \AUTHOR },}
